# Initialize and assign values to vec_x and vec_z
vec_x = np.random.uniform(low = 10, high = 20, size = 1000)
vec_z = np.random.uniform(low = -2, high = 5, size = 1000)

#create iterable list of vectors
vectors = [vec_x,vec_z]
#create iterable list of vector names
vectorNames = ["Vector X","Vector Z"]

#subplot creation
fig, sub_list = plt.subplots(1,2,figsize=(8,4))
plt.tight_layout() # to prevent overlap

# create histograms using for loop
ind = 0
for v in vectors:
    # histogram for vector
    sub_list[ind].hist(x=v)
    sub_list[ind].set_title(vectorNames[ind]+"'s Uniform Distribution")
    sub_list[ind].set_xlabel("Values")
    sub_list[ind].set_ylabel("Frequency")
    ind = ind + 1
